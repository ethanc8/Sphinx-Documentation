197 CF_INLINE CFIndex
198 GSUTF8CharacterAppend (UTF8Char * d, const UTF8Char * limit, UTF32Char c)
199 {
200  static const UTF8Char utf8LeadHeader[4] = { 0x00, 0xC0, 0xE0, 0xF0 };
201  CFIndex l;
202 
203  if (GSCharacterIsSurrogate (c))
204  return 0;
205 
206  l = GSUTF8CharacterLength (c);
207  if (l && limit - d > l)
208  {
209  switch (l)
210  {
211  case 4:
212  d[3] = (c & 0x3F) | 0x80;
213  c >>= 6;
214  case 3:
215  d[2] = (c & 0x3F) | 0x80;
216  c >>= 6;
217  case 2:
218  d[1] = (c & 0x3F) | 0x80;
219  c >>= 6;
220  case 1:
221  d[0] = c | utf8LeadHeader[l - 1];
222  }
223  }
224 
225  return l;
226 }

240 CF_INLINE CFIndex
241 GSUTF8CharacterGet (const UTF8Char * s, const UTF8Char * limit, UTF32Char *c)
242 {
243  static const UTF32Char utf8LeadMask[4] = { 0x0, 0x1F, 0x0F, 0x07 };
244  const UTF8Char *start;
245  UTF32Char ch;
246 
247  start = s;
248  ch = *s++;
249 
250  if (ch > 0x7F)
251  {
252  CFIndex trail;
253 
254  trail = GSUTF8CharacterTrailBytesCount (ch);
255  if (limit - s < trail)
256  trail = 0; /* Force an error */
257  ch &= utf8LeadMask[trail];
258  switch (trail)
259  {
260  case 3:
261  if (!GSUTF8CharacterIsTrailing (*s))
262  {
263  s = start;
264  break;
265  }
266  ch = (ch << 6) | (*s++ & 0x3F);
267  case 2:
268  if (!GSUTF8CharacterIsTrailing (*s))
269  {
270  s = start;
271  break;
272  }
273  ch = (ch << 6) | (*s++ & 0x3F);
274  case 1:
275  if (!GSUTF8CharacterIsTrailing (*s))
276  {
277  s = start;
278  break;
279  }
280  ch = (ch << 6) | (*s++ & 0x3F);
281  break;
282  case 0:
283  s = start;
284  break;
285  }
286  }
287  *c = ch;
288 
289  return s - start;
290 }

317 CF_INLINE CFIndex
318 GSUTF16CharacterAppend (UTF16Char * d, const UTF16Char * limit, UTF32Char c)
319 {
320  if (c <= 0xFFFF)
321  {
322  if ((limit - d) > 1)
323  *d = c;
324  return 1;
325  }
326  else if (c <= 0x10FFFF)

333  return 2;
334  }

336  return 0;

351 CF_INLINE CFIndex
352 GSUTF16CharacterGet (const UTF16Char * s, const UTF16Char * limit, UTF32Char *c)
353 {
354  const UTF16Char *start;
355  UTF32Char ch;
356 
357  start = s;
358  ch = *s++;
359 
360  if (GSCharacterIsSurrogate (ch))
361  {
362  if (GSCharacterIsLeadSurrogate (ch) && s < limit
363  && GSCharacterIsTrailSurrogate (*s))
364  ch = (ch << 10) + (*s++) - ((0xD7C0 << 10) + 0xDC00);
365  else
366  --s;
367  }
368  *c = ch;
369 
370  return s - start;
371 }
